from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from .models import Movie
from .serializers import MovieSerializer

class MoviePicker(APIView):
    def get(self, request):
        mood = request.query_params.get('mood', None)
        if mood:
            # Example logic to filter movies based on a 'mood'
            # You can define moods and match them to genres or tags
            if mood == "happy":
                movies = Movie.objects.filter(genres__icontains="Comedy")
            elif mood == "sad":
                movies = Movie.objects.filter(genres__icontains="Drama")
            else:
                movies = Movie.objects.all()

            serializer = MovieSerializer(movies, many=True)
            return Response(serializer.data, status=status.HTTP_200_OK)

        return Response({"error": "Mood not specified"}, status=status.HTTP_400_BAD_REQUEST)
